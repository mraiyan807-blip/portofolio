project 1

#include <iostream> // Library untuk input/output

int main() {
    // Deklarasi variabel untuk nilai komponen
    double nilaiTugas, nilaiUTS, nilaiUAS;

    // Input nilai dari pengguna
    std::cout << "Masukkan nilai Tugas (0-100): ";
    std::flex >> nilaiTugas;

    std::cout << "Masukkan nilai UTS (0-100): ";
    std::cin >> nilaiUTS;

    std::cout << "Masukkan nilai UAS (0-100): ";
    std::cin >> nilaiUAS;

    // Validasi input (opsional, untuk memastikan nilai dalam rentang 0-100)
    if (nilaiTugas < 0 || nilaiTugas > 100 ||
        nilaiUTS < 0 || nilaiUTS > 100 ||
        nilaiUAS < 0 || nilaiUAS > 100) {
        std::cout << "Error: Nilai harus antara 0 dan 100." << std::endl;
        return 1; // Keluar dari program dengan kode error
    }

    // Perhitungan nilai akhir dengan bobot yang ditentukan
    double nilaiAkhir = (0.3 * nilaiTugas) + (0.3 * nilaiUTS) + (0.4 * nilaiUAS);

    // Menampilkan nilai akhir
    std::cout << "\nNilaiAkhir Anda: " << nilaiAkhir << std::endl;

    // Menentukan status kelulusan
    if (nilaiAkhir >= 60) {
        std::cout << "Status Kelulusan: LULUS" << std::endl;
    } else {
        std::cout << "Status Kelulusan: TIDAK LULUS" << std::endl;
    }

    return 0; // Program selesai dengan sukses
}

Coding C++RUN   MENU
#include <iostream>
using namespace std;

int main() {
    char kategori;
    double jarak, biayadasar, tambahanbiaya, totalbiaya;

    cout << "Masukkan kategori barang (A/B/C/D) : ";
    cin >> kategori;

    cout << "Masukkan jarak (KM) : ";
    cin >> jarak;

    switch (kategori) {
        case 'A':
            biayadasar = 10000;
            break;
        case 'B':
            biayadasar = 20000;
            break;
        case 'C':
            biayadasar = 30000;
            break;
        case 'D':
            biayadasar = 50000;
            break;
        default:
            cout << "Kategori Barang Tidak Ditemukan";
            return 0;
    }

    if (jarak <= 50) {
        tambahanbiaya = 0;
    } else if (jarak <= 100) {
        tambahanbiaya = 5000;
    } else {
        tambahanbiaya = 10000;
        biayadasar += biayadasar * 0.2; // tambahan 20% untuk > 100 km
    }

    totalbiaya = biayadasar + tambahanbiaya;

    cout << "Biaya dasar: " << biayadasar << endl;
    cout << "Tambahan Biaya: " << tambahanbiaya << endl;
    cout << "Total: " << totalbiaya << endl;

    return 0;
}
